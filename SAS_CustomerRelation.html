CREATE TRIGGER [dbo].[gscTrig_RMCustomerRelationUpdate] ON [dbo].[SAS_CustomerRelation]
AFTER UPDATE, DELETE, INSERT
AS
BEGIN
   SET NOCOUNT ON;
   
 -- sync related organisations for which contact information has changed
 DECLARE @RelationId int, @OrganisationId varchar(50), @message varchar(max)
 
 WHILE 1=1
 BEGIN
   WITH CR as (
     -- items where relation became contact person entry
     -- when updated or created
     
     SELECT i.CustomerRelationId, i.OrganisationId, i.CustomerId
     FROM INSERTED i
       JOIN dbo.SAS_ListView lv ON i.RoleTypeId = lv.ListItemId and lv.CompanyId = i.CompanyId and lv.ListName = 'CUSTOMERROLE' and lv.ItemName = 'CONTACT PERSON'
       JOIN dbo.CustomerInformation ci ON ci.CustomerId = i.OrganisationID
     WHERE  isNull(i.ToDate, getdate()) <= getdate() -- organisation should have this person in contacts
        AND ci.IsIndividual = 0  -- sync only related ORGANISATION
        AND (ci.SAPID IS NOT NULL OR ci.RecordStatus = 1) -- don't create blocked related organisation
     -- previous relations
     UNION      
     SELECT d.CustomerRelationId, d.OrganisationId, d.CustomerId
     FROM DELETED d
       JOIN dbo.SAS_ListView lv ON d.RoleTypeId = lv.ListItemId and lv.CompanyId = d.CompanyId and lv.ListName = 'CUSTOMERROLE' and lv.ItemName = 'CONTACT PERSON'
       JOIN dbo.CustomerInformation ci ON ci.CustomerId = d.OrganisationID
     WHERE  isNull(d.ToDate, getdate()) <= getdate() -- organisation should have this person in contacts
        AND ci.IsIndividual = 0  -- sync only related ORGANISATION
        AND (ci.SAPID IS NOT NULL OR ci.RecordStatus = 1) -- don't create blocked related organisation
   )
 
   SELECT TOP 1
     @RelationId = CustomerRelationID, @OrganisationId = OrganisationID,
     @message = 'Put related organisation of customer ' + cast(CustomerId as varchar) + ' into synchronisation queue'
   FROM CR
   WHERE  CustomerRelationID > isNull(@RelationId, 0)
   IF @@ROWCOUNT = 0 BREAK;
   
     EXEC sap.CustomerWorker_Log
         @Operation = 'PUT',
         @EntityId = @OrganisationId,
         @EntityType = 'C',
         @ReturnMessage = @message
   
   EXEC dbo.SAS_DebtorUpsertCustomer @OrganisationId;          
 END


END

GO
