CREATE TRIGGER [dbo].[tr_SAS_WacUser_Add]
  ON  [dbo].[SAS_WacUser]
  AFTER INSERT,UPDATE
AS
BEGIN
   SET NOCOUNT ON;
   DECLARE @ChangeDateTime DATETIME2(4) = GETDATE();

   INSERT INTO dbo.SAS_WacUser_History (
       ChangedByUserName, ChangedDateTime,
       WacUserId, UserName, DisplayName,
       Email, Phone, MobilePhone, SiteId,
       PositionId, CompanyID, ActiveInCompany )
   SELECT
       ISNULL(I.ChangedByUserName, 'Support'), CASE WHEN D.WacUserId IS NULL THEN I.CreatedDate ELSE @ChangeDateTime END AS ChangeDateTime,
       I.WacUserId, I.UserName, I.DisplayName,
       I.Email, I.Phone, I.MobilePhone, I.SiteId,
       I.PositionId, NULL, NULL
   FROM INSERTED I
   JOIN (
           SELECT WacUserId, UserName, ISNULL(DisplayName,'') dn, ISNULL(Email,'') em, ISNULL(Phone,'') ph, ISNULL(MobilePhone,'') mp, SiteId, PositionId
           FROM INSERTED
           EXCEPT
           SELECT WacUserId, UserName, ISNULL(DisplayName,'') dn, ISNULL(Email,'') em, ISNULL(Phone,'') ph, ISNULL(MobilePhone,'') mp, SiteId, PositionId
           FROM DELETED   
   ) DetectChanges ON DetectChanges.WacUserId = I.WacUserId  -- Only audit updates that have a real effect
   LEFT JOIN DELETED D ON D.WacUserId = I.WacUserId
END
GO
