-- Description:    Create an audit trail of changes to users rights

CREATE TRIGGER [dbo].[tr_SAS_WacUserRight]
  ON  [dbo].[SAS_WacUser_Right]
  AFTER INSERT,UPDATE,DELETE
AS
BEGIN
   SET NOCOUNT ON;
   DECLARE @ChangeDateTime DATETIME2(4) = GETDATE();

   DECLARE @Action CHAR(1) = 'M';
   IF NOT EXISTS (SELECT * FROM INSERTED) SET @Action='D';
   IF NOT EXISTS (SELECT * FROM DELETED) SET @Action='A';

   INSERT INTO dbo.SAS_WacUser_Right_History (
       ChangedByUserName,
       ChangedDateTime,
       ChangedActionCode,
       WacUserID, CompanyID, ModuleID, AccessLevel )
   SELECT
       ISNULL(I.ChangedByUserName, 'Support'),
       @ChangeDateTime AS ChangedDateTime,
       @Action,
       I.WacUserID, I.CompanyID, I.ModuleID, I.AccessLevel
   FROM INSERTED I
   JOIN (
           SELECT WacUserID, CompanyID, ModuleID, AccessLevel
           FROM INSERTED
           EXCEPT
           SELECT WacUserID, CompanyID, ModuleID, AccessLevel
           FROM DELETED   
   ) DetectChanges ON  -- Only audit updates that have a real effect
       DetectChanges.WacUserID = I.WacUserID
       AND DetectChanges.CompanyID = I.CompanyID
       AND DetectChanges.ModuleID = I.ModuleID;

   IF (@Action = 'D')
   BEGIN
     INSERT INTO dbo.SAS_WacUser_Right_History (
       ChangedByUserName,
       ChangedDateTime,
       ChangedActionCode,
       WacUserID, CompanyID, ModuleID, AccessLevel )
   SELECT
       ISNULL(D.ChangedByUserName, 'Support'),
       @ChangeDateTime AS ChangedDateTime,
       @Action,
       D.WacUserID, D.CompanyID, D.ModuleID, D.AccessLevel
   FROM DELETED D  
   END  

END
GO
