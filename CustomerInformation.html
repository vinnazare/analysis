CREATE TRIGGER [dbo].[gscTrig_RMCustomerInsert] ON [dbo].[CustomerInformation]
FOR INSERT
AS
BEGIN
   SET NOCOUNT ON;
   DECLARE @CustomerID            nvarchar(50);

   SELECT @CustomerID=CustomerID
   FROM Inserted
   WHERE IsIndividual = 0;

   IF @CustomerID IS NOT NULL
   BEGIN
   
       EXEC dbo.SAS_DebtorUpsertCustomer @CustomerID;
   END
END
GO

    OR UPDATE(MailingCountry) OR UPDATE(MailingState) OR UPDATE(MailingSuburb) OR UPDATE(MailingPostcode) OR UPDATE(MailingLocality) OR UPDATE(MailingRoadNo) OR UPDATE(MailingRoadName) OR UPDATE(MailingCareOf)
    OR UPDATE(ABN) OR UPDATE(Mobile)
    OR UPDATE(CompanyId)
 BEGIN
     DECLARE @CustomerID    VARCHAR(50);

   WHILE 1=1
   BEGIN
       SELECT TOP 1 @CustomerID = I.CustomerID  
       FROM INSERTED I
       JOIN DELETED D    
           ON D.CustomerID = I.CustomerId
       WHERE
           I.CustomerID > ISNULL(@CustomerID,'')      
           AND I.IsIndividual = 0 -- don't sync individual
           AND (I.SAPID IS NOT NULL OR I.RecordStatus = 1) -- don't create blocked
           AND (
               I.RecordStatus<>D.RecordStatus OR
               ISNULL(I.CustomerName,'')<>ISNULL(D.CustomerName,'') OR
               ISNULL(I.ResCountry,'')<>ISNULL(D.ResCountry,'') OR
               ISNULL(I.ResState,'')<>ISNULL(D.ResState,'') OR
               ISNULL(I.ResSuburb,'')<>ISNULL(D.ResSuburb,'') OR
               ISNULL(I.ResPostcode,'')<>ISNULL(D.ResPostcode,'') OR
               ISNULL(I.ResLocality,'')<>ISNULL(D.ResLocality,'') OR
               ISNULL(I.ResRoadNo,'')<>ISNULL(D.ResRoadNo,'') OR
               ISNULL(I.ResRoadName,'')<>ISNULL(D.ResRoadName,'') OR
               ISNULL(I.CustomerPhone,'')<>ISNULL(D.CustomerPhone,'') OR
               ISNULL(I.CustomerFax,'')<>ISNULL(D.CustomerFax,'') OR
               ISNULL(I.CustomerEmail,'')<>ISNULL(D.CustomerEmail,'') OR
               ISNULL(I.MailingCountry,'')<>ISNULL(D.MailingCountry,'') OR
               ISNULL(I.MailingState,'')<>ISNULL(D.MailingState,'') OR
               ISNULL(I.MailingSuburb,'')<>ISNULL(D.MailingSuburb,'') OR
               ISNULL(I.MailingPostcode,'')<>ISNULL(D.MailingPostcode,'') OR
               ISNULL(I.MailingLocality,'')<>ISNULL(D.MailingLocality,'') OR
               ISNULL(I.MailingRoadNo,'')<>ISNULL(D.MailingRoadNo,'') OR
               ISNULL(I.MailingRoadName,'')<>ISNULL(D.MailingRoadName,'') OR
               ISNULL(I.MailingCareOf,'')<>ISNULL(D.MailingCareOf,'') OR
               ISNULL(I.ABN,'')<>ISNULL(D.ABN,'') OR
               ISNULL(I.CompanyId,'')<>ISNULL(D.CompanyId,'') OR
               ISNULL(I.Mobile,'')<>ISNULL(D.Mobile,'')
           );
       IF @@ROWCOUNT = 0 BREAK;
     
       IF @CustomerID IS NOT NULL
           EXEC dbo.SAS_DebtorUpsertCustomer @CustomerID;
   END
 END



 -- sync related organisations for which contact information has changed
 IF UPDATE(RecordStatus) OR UPDATE(CustomerSalutation) OR UPDATE(CustomerFirstName) OR UPDATE(CustomerLastName) OR UPDATE(CustomerPhone) OR UPDATE(HomePhone) OR UPDATE(Mobile) OR UPDATE(CustomerEmail)
 BEGIN
   DECLARE @RelationId INT, @OrganisationId VARCHAR(50), @message VARCHAR(MAX);

     DECLARE @ContactRoleTypeID INT;
     SELECT TOP 1 @ContactRoleTypeID=ListItemId
     FROM dbo.SAS_ListView lv
     JOIN INSERTED I ON I.CompanyID=lv.CompanyId
     WHERE lv.ListName = 'CUSTOMERROLE' AND lv.ItemName = 'CONTACT PERSON';    -- Assume same list item id for all companies!!
   
   WHILE 1=1
   BEGIN
     SELECT TOP 1
       @RelationId = CustomerRelationID, @OrganisationId = OrganisationID,
       @message = 'Put related organisation of customer ' + CAST(I.CustomerId AS VARCHAR) + ' into synchronisation queue'
     FROM INSERTED    I
     JOIN dbo.SAS_CustomerRelation cr
       ON cr.CustomerID = I.CustomerId AND cr.RoleTypeID = @ContactRoleTypeID
     JOIN dbo.CustomerInformation ci
       ON ci.CustomerId = cr.OrganisationID
     JOIN DELETED D
       ON D.CustomerID = I.CustomerId
     WHERE  cr.CustomerRelationID > ISNULL(@RelationId, 0)
       AND ISNULL(cr.ToDate, GETDATE()) <= GETDATE() -- organisation still should have this person in contacts
       AND ci.IsIndividual = 0  -- sync only related ORGANISATION
       AND (ci.SAPID IS NOT NULL OR ci.RecordStatus = 1) -- don't create blocked related organisation
       AND (
               I.RecordStatus<>D.RecordStatus OR
               ISNULL(I.CustomerSalutation,'')<>ISNULL(D.CustomerSalutation,'') OR
               ISNULL(I.CustomerFirstName,'')<>ISNULL(D.CustomerFirstName,'') OR
               ISNULL(I.CustomerLastName,'')<>ISNULL(D.CustomerLastName,'') OR
               ISNULL(I.CustomerPhone,'')<>ISNULL(D.CustomerPhone,'') OR
               ISNULL(I.HomePhone,'')<>ISNULL(D.HomePhone,'') OR
               ISNULL(I.Mobile,'')<>ISNULL(D.Mobile,'') OR
               ISNULL(I.CustomerEmail,'')<>ISNULL(D.CustomerEmail,'')
           )
     IF @@ROWCOUNT = 0 BREAK;
     
         EXEC sap.CustomerWorker_Log
             @Operation = 'PUT',
             @EntityId = @OrganisationId,
             @EntityType = 'C',
             @ReturnMessage = @message
     
     EXEC dbo.SAS_DebtorUpsertCustomer @OrganisationId;          
   END
 END

END
