CREATE TRIGGER [dbo].[tr_SAS_WacUser_Company_Add]
  ON  [dbo].[SAS_WacUser_Company]
  AFTER INSERT,UPDATE
AS
BEGIN
   SET NOCOUNT ON;
   DECLARE @ChangeDateTime DATETIME2(4) = GETDATE();

   INSERT INTO dbo.SAS_WacUser_History (
       ChangedByUserName,
       ChangedDateTime,
       WacUserId, UserName, DisplayName, CompanyID, ActiveInCompany )
   SELECT
       ISNULL(I.ChangedByUserName, 'Support'),
       @ChangeDateTime AS ChangedDateTime,
       I.WacUserId, U.UserName, U.DisplayName,
       I.CompanyID, I.Active
   FROM INSERTED I
   JOIN (
           SELECT UserID, CompanyId, Active
           FROM INSERTED
           EXCEPT
           SELECT UserID, CompanyId, Active
           FROM DELETED   
   ) DetectChanges ON DetectChanges.UserID=I.UserID  -- Only audit updates that have a real effect
   LEFT JOIN SAS_WacUser U ON U.WacUserId = I.WacUserId

END
GO
