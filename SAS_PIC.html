CREATE TRIGGER [dbo].[trg_PICHist] ON [dbo].[SAS_PIC]
FOR UPDATE


AS

IF @@RowCount > 1 RETURN

declare @OldManagerId varchar(50)
declare @OldTradingNameId varchar(50)
declare @OldPICStatusId int

declare @NewManagerId varchar(50)
declare @NewTradingNameId varchar(50)
declare @NewPICStatusId int

SELECT @OldManagerId = ManagerId, @OldTradingNameId = TradingNameId, @OldPICStatusId = PICStatusId FROM Deleted


SELECT @NewManagerId = ManagerId, @NewTradingNameId = TradingNameId, @NewPICStatusId = PICStatusId FROM Inserted

IF @OldManagerId <> @NewManagerId OR isnull(@OldTradingNameId,0) <> isnull(@NewTradingNameId,0) OR @NewPICStatusId <> @OldPICStatusId
BEGIN

   IF @NewPICStatusId <> @OldPICStatusId
   BEGIN
       INSERT INTO SAS_PIC_Hist(PICId, ManagerId, TradingNameId, PICStatusId, ModifiedBy, ModifiedDate, StatusChangeReasonId, NewPICStatusID, NewManagerId, NewTradingNameId, StatusChangeDate)
       SELECT D.PICId, D.ManagerId, D.TradingNameID, D.PICStatusID, I.ModifiedBy, I.ModifiedDate, I.StatusChangeReasonId, I.PICStatusId, I.ManagerId, I.TradingNameId, I.StatusChangeDate
       FROM Deleted D, Inserted I WHERE D.PICId = I.PICId
   END
   ELSE
   BEGIN
   INSERT INTO SAS_PIC_Hist(PICId, ManagerId, TradingNameId, PICStatusId, ModifiedBy, ModifiedDate, NewManagerId, NewTradingNameId)
       SELECT D.PICId, D.ManagerId, D.TradingNameID, D.PICStatusID, I.ModifiedBy, I.ModifiedDate, I.ManagerId, I.TradingNameId
       FROM Deleted D, Inserted I WHERE D.PICId = I.PICId
   END
END








GO
